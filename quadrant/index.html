        .attr("width", 700)

        .enter()

        var linearScale = d3.scale.linear()
        .domain([0,100])
        .range([0,600]);



    

            group.append("rect")
            .attr("width", function(d){
            return linearScale(d);
            })
            .attr("height", 30)
            .attr("x", 0)
            .attr("y", function(d,i){
            return i*40 ;
            })
            .style("fill", function(d) { return "white"; });
            group.append("text")
            .attr("x", function(d){
            return linearScale(d);
            })
            .attr("y", function(d,i){
            return i*40 ;
            })
            .attr("dy", +30)
            .text( function (d) { return d+"%"; })

X,Y軸建立

程式碼



                var svg = d3.select("#aaa6")
                .attr("width", 1000)
                .attr("height", 600);

                svg.append("rect")
                .attr("width", "100%")
                .attr("height", "100%")
                .attr("fill", "pink");

                var x = d3.scale.linear().domain([0, 10]).range([0, 800]);
                var y = d3.scale.linear().domain([0, 10]).range([500, 50]);

                var axis = d3.svg.axis()
                .scale(x);

                var a = svg.append('g')
                .attr('transform', "translate(100, +"+(500)+")")
                .attr("class", "xaxis")
                .call(axis);

                var yaxis = d3.svg.axis()
                .orient('left')
                .scale(y);

                var ya = svg.append('g')
                .attr('transform', 'translate('+100+', 0)')
                .call(yaxis);
