D3.js group添附circle

append X,Y軸到畫布上 


    circle1.append("circle")
    .data(data1)
    .style('fill', function (d) {
        return d.color;
    })
    .attr("r", function (d) {
        return d.value / 100000;
    })
    .style("stroke", function (d) {
        return d.color;
    })
