        var svg = d3.select("#graph").append("svg")

線性縮放函數

建立X,Y線性縮放函數 


        var x = d3.scale.linear()
    .domain([-max_domoin_X, max_domoin_X])
    .range([svg_left_margin, width - margin]);

var y = d3.scale.linear()
    .domain([-max_domoin_Y, max_domoin_Y])
    .range([height - margin, margin]);

X,Y軸建立

建立X,Y軸,並使用前頁之線性縮放函數 


       var axis = d3.svg.axis()
    .scale(x)
    .ticks(max_domoin_X*2)
    .tickFormat(function(d) { return xlabel[d+max_domoin_X] });
var yaxis = d3.svg.axis()
    .orient('left')
    .ticks(max_domoin_Y*2)
    .tickFormat(function(d) { return ylabel[d+max_domoin_Y] })
    .scale(y);

    svg.append('g')

畫布添附circle群組

畫布新增一circle群組, 主要用來統一管理所有circles 


    var circle1 = svg.selectAll("g.circle1")
                        .data(data)
                        .enter()
                        .append("g")
                        .attr("class", "circle1");

circle群組添附circles

將circles添附到circle群組裡 


    circle1.append("circle")
    .data(data)
    .style('fill', function (d) {
        return d.color;
    })
    .attr("r", function (d) {
        return d.value / 100000;
    })
    .style("stroke", function (d) {
        return d.color;
    })

circle添附url

點選前往新網址 


    circle1
    .on("click", function (d) {
        //url = "https://google.com
        url = d.url;
        window.location = url;
    })

circle添附text

在circle下放置text tooltip 


    circle1.append("text")

                        .attr("x", function (d) {
                            return x(d.x);
                        })
                        .attr("y", function (d) {
                            return y(d.y) - 50;
                        })
                        .attr("dy", "1.25em")
                        .style('opacity', 0.0)
                        .attr("text-anchor", "middle")
                        .style('fill', 'red')
                        .attr("pointer-events", "none")
                        .attr("class", "tooltip_label")
                        .text(function (d) {
                            return d.label;
                        });
